<p>  // Utility: sanitize text for HTML<br />
  function escapeHtml(str) {<br />
    if (!str) return '';<br />
    return str.replace(/[&<>"']/g, (m) => {<br />
      return ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;" })[m];<br />
    });<br />
  }</p>

<p>  // Render stars: supports decimals for 4.8 rating<br />
  function renderStars(n) {<br />
    const filledStars = Math.floor(n);<br />
    const partialStar = n - filledStars;<br />
    let starsHTML = '';<br />
    for (let i = 0; i < filledStars; i++) starsHTML += '★';<br />
    if (partialStar >= 0.75) starsHTML += '★';<br />
    else if (partialStar >= 0.25) starsHTML += '⯪'; // partial star alternative (optional)<br />
    const emptyStars = 5 - starsHTML.length;<br />
    for (let i = 0; i < emptyStars; i++) starsHTML += '☆';<br />
    return `<span style="color:#ffd86b">${starsHTML}</span><span style="color:rgba(255,255,255,0.12)">${'☆'.repeat(emptyStars)}</span>`;<br />
  }</p>

<p>  // Calculate and update stats display<br />
  function updateStats() {<br />
    const allRatings = countOnly.concat(written.map(r => r.rating));<br />
    const totalCount = allRatings.length;<br />
    const avg = totalCount ? (allRatings.reduce((a, b) => a + b, 0) / totalCount) : 0;<br />
    avgStarsEl.textContent = totalCount ? avg.toFixed(1) + ' ★' : '—';<br />
    countEl.textContent = totalCount;<br />
  }</p>

<p>  // Build ticker content from written reviews, duplicated for smooth scroll<br />

      return;<br />
    }<br />
    const nodes = written.map(r => {<br />
      return `<div class="item"><br />
        <div class="who">${escapeHtml(r.name)}</div><br />
        <div class="stars">${renderStars(r.rating)}</div><br />
        <div class="txt">${escapeHtml(r.text)}</div><br />

    });<br />
    // Duplicate for seamless scroll<br />
    tickerEl.innerHTML = nodes.join('') + nodes.join('');<br />
  }</p>
